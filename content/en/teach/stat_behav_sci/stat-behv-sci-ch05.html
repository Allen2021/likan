1 Figure.05.02

X <- function(mean, sd) {
     X <- seq((mean - 3 * sd),
              (mean + 3 * sd),
              by = 0.01
             )
     return(X)
}

Y <- function (mean, sd, ...) {
      dots <- list(...)
      if (length(dots) == 0)
         X <- X(mean, sd)
      else
        X <- unlist(dots)
      Y <- dnorm(X,
                mean = mean,
                sd = sd
           )
      return(Y)
}

pp <- function(mean, sd){
plot(X(mean, sd),
     Y(mean, sd),
     type = "l",
     xlab = "",
     ylab = "",
     axes = FALSE,
     col = "#af1e23",
     lwd = 2
)
abline(h = 0)
segments(
     c(mean, mean + sd),
     c(0, 0),
     c(mean, mean + sd),
     c(Y(mean, sd, mean),
       Y(mean, sd, mean + sd)
      ),
      lty = c(1, 2)
)
arrows(
      mean,
      Y(mean, sd, mean + sd),
      mean + sd,
      Y(mean, sd, mean + sd),
      length = 0.08
)
text(
    x = mean + sd / 2,
    y = Y(mean, sd, mean + sd),
    pos = 3,
    label = bquote(sigma == .(sd)),
)
mtext(c(mean, mean + sd),
      side = c(1, 1),
      adj = c(3, 4) / 6,
      padj = - 0.7
)
mtext("X = 76",
      side = 1,
      adj = (3 + 6 / sd ) /6,
      padj = 2
)
arrows(76,
      -Y(mean, sd, mean + sd) / 6,
      76,
      0,
      xpd = TRUE,
      length = 0.08
)
}
par(mfrow = c(1, 2))
pp(70, 3)
pp(70, 12)

4 Centering and scale

X <- c(0, 2, 4, 4, 5)
SS <- sum(X^2) - (sum(X))^2 / length(X)
s <- sqrt(SS / (length(X) - 1))
M <- mean(X)
z <- (X - M) / s
print(z)
